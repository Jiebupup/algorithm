    public ArrayList<Integer> PrintFromTopToBottom(TreeNode root) {
        ArrayList<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            TreeNode p = q.poll();
            if (p.left != null)
                q.offer(p.left);
            if (p.right != null)
                q.offer(p.right);
            res.add(p.val);
        }
        return res;
    }

    public ArrayList<Integer> PrintFromTopToBottom(TreeNode root) {
        ArrayList<Integer> res = new ArrayList<>();
        if (root == null)
            return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        while (!q.isEmpty()) {
            int cnt = q.size();
            while (cnt-- > 0) {
                TreeNode p = q.poll();
                if (p == null)
                    continue;
                res.add(p.val);
                q.offer(p.left);
                q.offer(p.right);
            }
        }
        return res;
    }
