# 方法一：diff 为不存在重复的两个元素异或的结果，diff &= -diff 得到不存在重复的两个元素在位级表示上最右侧不同的那一位，利用这一位就可以将两个元素区分开来

    public void FindNumsAppearOnce(int[] array, int num1[], int num2[]) {
        int diff = 0;
        for (int a : array) {
            diff ^= a;
        }
        diff &= -diff;
        for (int a : array) {
            if ((diff & a) == 0)
                num1[0] ^= a;
            else
                num2[0] ^= a;
        }
    }
