    public boolean IsPopOrder(int[] pushA, int[] popA) {
        if (pushA == null || pushA.length == 0)
            return false;
        Stack<Integer> s = new Stack<>();
        for (int i = 0, j = 0; i < pushA.length; i++) {
            s.push(pushA[i]);
            while (!s.isEmpty() && s.peek() == popA[j]) {
                s.pop();
                j++;
            }
        }
        return s.isEmpty();
    }
