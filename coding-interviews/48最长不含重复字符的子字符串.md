    public String PrintMinNumber(int[] numbers) {
        int len = numbers.length;
        if (numbers == null || len == 0)
            return "";
        String[] sa = new String[len];
        for (int i = 0; i < len; i++) {
            sa[i] = numbers[i] + "";
        }
        Arrays.sort(sa, (s1, s2) -> (s1 + s2).compareTo(s2 + s1));
        StringBuilder res = new StringBuilder();
        for (String s : sa) {
            res.append(s);
        }
        return res.toString();
    }
