## 方法一：依旧斐波那契数列

    public int JumpFloor(int target) {
        if (target <= 2)
            return target;
        int res = 0;
        int a = 1;
        int b = 2;
        for (int i = 3; i <= target; i++) {
            res = a + b;
            a = b;
            b = res;
        }
        return res;
    }
